export const personalizedOutreach = defineGenerativeFunction({

Write a highly personalized cold email that:

   - Reference something SPECIFIC and RECENT about them or their company
   - Options to consider:
     * Recent LinkedIn post they wrote
     * Company announcement or milestone
     * Recent hire in their team
     * GitHub project they contributed to
     * Blog post they published
   - Make it feel like genuine recognition, not flattery

2. CONTEXT (1-2 sentences):
   - Connect their situation to the problem we solve
   - Show you understand their specific challenge
   - Use their terminology and technical level

3. VALUE PROPOSITION (2-3 sentences):
   - Explain how sales-builder solves THEIR specific problem
   - Use a concrete example relevant to their industry/stage
   - Include specific data point (e.g., "Companies like yours increase
     outbound by 3-4x without hiring")

4. PROOF POINT (1 sentence):
   - Brief customer example in similar situation
   - Or relevant result/metric

5. CALL TO ACTION (1 sentence):
   - Low-friction ask (not "buy now")
   - Offer specific value in exchange for time
   - Example: "Want to see how [Company X] implemented this?"

REQUIREMENTS:
- Max 150 words total
- Conversational, peer-to-peer tone (not salesy)
- Technical enough to show competence, not so much to confuse
- Include one thoughtful question
- Natural, not obviously templated
- NO generic praise or flattery
- NO buzzwords or hype language

PERSONALIZATION SCORE TARGET: 9/10
(Score measures: specificity, relevance, authenticity, value)`,

  // Quality validation
  validation: {
    checks: [
      {
        name: 'personalization-depth',
        evaluate: async (email) => {
          const specificReferences = countSpecificReferences(email)
          const genericPhrases = countGenericPhrases(email)

          return {
            score: specificReferences / (specificReferences + genericPhrases),
            passed: specificReferences >= 2 && genericPhrases === 0
          }
        }
      },

      {
        name: 'length-check',
        evaluate: (email) => {
          const wordCount = countWords(email)
          return {
            score: wordCount <= 150 ? 1.0 : 150 / wordCount,
            passed: wordCount <= 150
          }
        }
      },

      {
        name: 'authenticity',
        evaluate: async (email) => {
          const flags = detectInauthenticLanguage(email, [
            'hope this email finds you well',
            'reaching out to',
            'i wanted to touch base',
            'thought you might be interested',
            'hope you're doing well',
            'quick question',
            'just following up'
          ])

          return {
            score: 1 - (flags.length / 10),
            passed: flags.length === 0
          }
        }
      },

      {
        name: 'value-clarity',
        evaluate: async (email) => {
          const hasSpecificValue = /\d+x|[0-9]+%|[0-9]+ hours/.test(email)
          const hasVagueValue = /(improve|better|optimize|enhance) (?!by \d)/.test(email)

          return {
            score: hasSpecificValue ? 1.0 : 0.5,
            passed: hasSpecificValue
          }
        }
      }
    ],

    minScore: 0.85, // Must pass 85% of checks

    onFail: {
      action: 'regenerate',
      maxAttempts: 3,
      escalateAfter: 3
    }
  }
})

## Personalization Layers: From Generic to Hyper-Personal

AI applies personalization in layers:

### Layer 1: Basic (Everyone Gets This)
- Name
- Company
- Title
- Industry

### Layer 2: Company Context
- Company size
- Funding stage
- Growth signals
- Tech stack

### Layer 3: Role-Specific
- Challenges specific to their role
- Relevant use cases
- Appropriate technical depth

### Layer 4: Recent Activity
- LinkedIn posts
- Company announcements
- Hiring signals
- Product launches

### Layer 5: Hyper-Personal
- Specific LinkedIn post content
- GitHub project details
- Article they wrote
- Conference they spoke at
- Shared connections

Higher layers = higher conversion, but require more data:

```typescript
const determinePersonalizationLayer = (prospect) => {
  const data = prospect.enrichment

  if (data.linkedinPosts?.length > 0 || data.githubActivity?.length > 0) {
    return 5 // Hyper-personal
  }

  if (data.companyNews?.length > 0 || data.signals?.length > 0) {
    return 4 // Recent activity
  }

  if (data.title && data.challenges?.length > 0) {
    return 3 // Role-specific
  }

  if (data.companySize && data.techStack?.length > 0) {
    return 2 // Company context
  }

  return 1 // Basic
}

const applyPersonalization = async (prospect, layer) => {
  switch (layer) {
    case 5:
      return await generateHyperPersonalEmail(prospect)
    case 4:
      return await generateActivityBasedEmail(prospect)
    case 3:
      return await generateRoleSpecificEmail(prospect)
    case 2:
      return await generateCompanyContextEmail(prospect)
    case 1:
      return await generateBasicPersonalizedEmail(prospect)
  }
}
```
