# The 24/7 SDR: AI Agents That Never Sleep, Never Miss a Lead

## The SDR Economics Problem

Hiring human SDRs has never made economic sense for early-stage startups:

**Human SDR Costs:**
- Salary: $60K-$80K base + $20K-$40K commission
- Benefits: $15K-$25K (health, 401k, etc.)
- Tools & training: $5K-$10K annually
- Management overhead: 20-30% of salary
- Ramp time: 3-6 months to productivity
- **Total first-year cost: $120K-$180K per SDR**

**Human SDR Output:**
- Working hours: 40/week = 2,080/year (minus vacation, sick days)
- Productive hours: ~60% (1,248 hours annually)
- Emails per day: 40-60
- Calls per day: 20-30
- Meetings booked: 4-8 per month
- **Total: ~50-100 meetings per year per SDR**

**Cost per meeting booked: $1,200-$3,600**

For a startup trying to reach $1M ARR, this math doesn't work.

## The AI SDR Alternative

```typescript
import { defineAISDR } from '@sales-builder/agents'

export const aiSDR = defineAISDR({
  // ICP definition
  targeting: {
    titles: ['Founder', 'CTO', 'VP Engineering', 'Head of Product'],
    companySize: [1, 100],
    industries: ['SaaS', 'Developer Tools', 'AI/ML'],
    technographics: ['uses:typescript', 'uses:react', 'uses:aws'],
    signals: [
      'product-hunt-launch',
      'funding-announcement',
      'hiring-engineers',
      'techcrunch-mention'
    ]
  },

  // Lead sourcing
  sourcing: {
    sources: ['apollo', 'clearbit', 'hunter', 'linkedin-sales-nav'],
    dailyNewLeads: 50,
    enrichment: {
      company: ['revenue', 'headcount', 'tech-stack', 'funding'],
      person: ['email-verified', 'linkedin-url', 'recent-activity']
    }
  },

  // Qualification logic
  qualification: {
    scoreFactors: {
      titleMatch: { weight: 25, score: (title) => exactMatch(title, targeting.titles) },
      companySize: { weight: 15, score: (size) => inRange(size, targeting.companySize) },
      industryMatch: { weight: 15, score: (industry) => matches(industry, targeting.industries) },
      hasSignal: { weight: 20, score: (signals) => hasAny(signals, targeting.signals) },
      techStack: { weight: 15, score: (tech) => overlap(tech, targeting.technographics) },
      engagement: { weight: 10, score: (history) => calculateEngagement(history) }
    },

    thresholds: {
      cold: { score: { max: 40 }, action: 'nurture-sequence' },
      warm: { score: { min: 41, max: 69 }, action: 'standard-outreach' },
      hot: { score: { min: 70 }, action: 'priority-outreach-immediate' }
    }
  },

  // Outreach strategy
  outreach: {
    channels: ['email', 'linkedin'],

    email: {
      dailyLimit: 100,
      sequences: {
        cold: {
          steps: [
            { day: 0, template: 'initial-value-prop' },
            { day: 3, template: 'case-study-relevant', condition: 'not:replied' },
            { day: 7, template: 'different-angle', condition: 'not:replied' },
            { day: 14, template: 'breakup-email', condition: 'not:replied' }
          ]
        },

        warm: {
          steps: [
            { day: 0, template: 'personalized-insight' },
            { day: 4, template: 'specific-use-case', condition: 'not:replied' },
            { day: 10, template: 'final-touch', condition: 'not:replied' }
          ]
        },

        hot: {
          steps: [
            { day: 0, template: 'highly-personalized-immediate-value' },
            { day: 2, template: 'founder-direct-outreach', condition: 'not:replied' }
          ]
        }
      },

      personalization: {
        model: 'claude-opus-4.5',
        dataPoints: [
          'recent-company-news',
          'recent-linkedin-posts',
          'github-activity',
          'shared-connections',
          'tech-stack-analysis',
          'competitive-intel'
        ],
        tone: 'peer-to-peer-technical',
        length: { max: 150 }, // words
        includeQuestion: true
      }
    },

    linkedin: {
      connectionsPerDay: 20,
      sequence: [
        { action: 'send-connection', note: 'personalized-1-sentence' },
        { day: 1, action: 'send-message', condition: 'accepted', template: 'linkedin-intro' },
        { day: 4, action: 'send-message', condition: 'not:replied', template: 'linkedin-follow-up' }
      ]
    }
  },

  // Response handling
  responseHandling: {
    autoRespond: true,

    responseTypes: {
      positive: {
        triggers: ['interested', 'tell me more', 'yes', 'schedule', 'when'],
        action: async ({ lead, conversation }) => {
          return {
            response: await generateResponse({
              type: 'book-meeting',
              include: 'calendar-link',
              tone: 'helpful-not-pushy'
            }),
            updateStage: 'qualified',
            notifyHuman: true,
            priority: 'high'
          }
        }
      },

      questions: {
        triggers: ['how', 'what', 'does it', 'can you'],
        action: async ({ lead, question }) => {
          const answer = await generateAnswer({
            question: question,
            context: lead,
            knowledgeBase: 'product-docs-and-faqs',
            includeProof: true
          })

          return {
            response: answer,
            addToKnowledgeBase: question.isNew,
            updateEngagement: +5
          }
        }
      },

      objections: {
        triggers: ['too expensive', 'not now', 'already have', 'not interested'],
        action: async ({ lead, objection }) => {
          const analysis = await analyzeObjection(objection)

          if (analysis.overcomeable) {
            return {
              response: await generateObjectionResponse({
                objection: objection,
                approach: analysis.recommendedApproach,
                includeProof: true
              }),
              updateStage: 'objection-handling'
            }
          } else {
            return {
              response: 'respectful-acceptance',
              updateStage: 'not-interested',
              addToNurture: { delay: '90d' }
            }
          }
        }
      },

      outOfOffice: {
        triggers: ['out of office', 'on vacation', 'parental leave'],
        action: async ({ lead, autoReply }) => {
          const returnDate = extractDate(autoReply)

          return {
            response: null, // Don't reply to auto-replies
            snooze: { until: returnDate },
            notifyHuman: false
          }
        }
      },

      escalation: {
        triggers: ['speak to founder', 'legal question', 'enterprise pricing', 'partnership'],
        action: async ({ lead, message }) => {
          return {
            response: await generateResponse({
              type: 'human-will-follow-up',
              timeline: '24-hours'
            }),
            escalateToHuman: true,
            priority: 'high',
            context: { lead, fullConversation: await getHistory(lead) }
          }
        }
      }
    },

    responseSpeed: {
      businessHours: { min: 2, max: 15 }, // minutes
      afterHours: { min: 5, max: 30 }, // minutes
      weekend: { min: 10, max: 60 } // minutes
    }
  },

  // Meeting booking
  meetingBooking: {
    automated: true,
    calendar: 'founder-calendar',

    qualification: {
      required: [
        'email-verified',
        'lead-score-above-60',
        'positive-engagement'
      ],
      skipQualification: {
        leadScore: { min: 85 },
        hasSignal: ['funding-announcement', 'urgent-need']
      }
    },

    scheduling: {
      provider: 'cal.com',
      meetingType: 'product-demo-30min',
      bufferDays: { min: 1, max: 14 }, // Book 1-14 days out
      reminderSequence: [
        { timing: '-24h', channel: 'email', template: 'meeting-reminder' },
        { timing: '-2h', channel: 'email', template: 'meeting-starting-soon' }
      ],
      noShowSequence: [
        { timing: '+30m', channel: 'email', template: 'we-missed-you' },
        { timing: '+2d', channel: 'email', template: 'reschedule-offer' }
      ]
    }
  },

  // Performance monitoring
  monitoring: {
    metrics: [
      'emails-sent',
      'emails-opened',
      'emails-replied',
      'meetings-booked',
      'meetings-held',
      'pipeline-created',
      'deals-closed'
    ],

    alerts: [
      { metric: 'reply-rate', below: 0.05, action: 'review-templates' },
      { metric: 'bounce-rate', above: 0.03, action: 'check-email-quality' },
      { metric: 'unsubscribe-rate', above: 0.02, action: 'review-messaging' },
      { metric: 'meeting-show-rate', below: 0.60, action: 'improve-qualification' }
    ],

    optimization: {
      abTest: {
        subjects: true,
        templates: true,
        timing: true,
        cadence: true
      },
      autoOptimize: true, // AI learns from results
      reportingFrequency: 'weekly'
    }
  }
})
```
